// SPDX-License-Identifier: MIT pragma solidity ^0.8.19; contract CoopEnergie { // ── État de la coopérative ── uint256 public objectif = 250_000; // FCFA uint256 public totalCollecte = 0; uint8 public quorum = 4; // sur 8 membres bool public achatAutorise = false; string public coopName; // ── Mappings immuables ── mapping(address => uint256) public contributions; mapping(address => bool) public aVote; mapping(uint8 => uint256) public votesOption; // ── Événements on-chain ── event CotisationEnregistree( address indexed membre, uint256 montant, uint256 timestamp ); event VoteEnregistre( address indexed votant, uint8 choix, uint256 timestamp ); event AchatAutorise(uint8 choixGagnant); // ── Cotiser ── function cotiser() external payable { require(msg.value > 0, "Montant nul"); contributions[msg.sender] += msg.value; totalCollecte += msg.value; emit CotisationEnregistree( msg.sender, msg.value, block.timestamp ); if (totalCollecte >= objectif) _verifierObjectif(); } // ── Voter ── function voter(uint8 choix) external { require(!aVote[msg.sender], "Déjà voté"); require(choix <= 2, "Option invalide"); aVote[msg.sender] = true; votesOption[choix]++; emit VoteEnregistre( msg.sender, choix, block.timestamp ); _verifierQuorum(); } // ── Vérifications automatiques ── function _verifierObjectif() internal { if (totalCollecte >= objectif) { achatAutorise = true; } } function _verifierQuorum() internal { uint256 total = votesOption[0] + votesOption[1] + votesOption[2]; if (total >= quorum) { uint8 gagnant = _calculerGagnant(); emit AchatAutorise(gagnant); } } }

⛓ Composante blockchain

Comment la blockchain garantit la transparence

Explication complète de l'architecture blockchain de CoopEnergie — comment les cotisations sont tracées, comment le vote est sécurisé, comment la transparence totale est garantie.

🔗 Fonctionnement de la chaîne de blocs
1
Création du bloc genesis
À la création de la coopérative, un premier bloc inviolable est inscrit. Il contient le nom, l'objectif, les membres et les règles de vote. Ce bloc est la fondation immuable de toute la chaîne.
Block #1 | type: GENESIS | prevHash: 0x000…000 | hash: 0x4f2a…c8b1
2
Chaque cotisation = un bloc immuable
Quand un membre cotise, un bloc est créé contenant son nom, le montant, le mode de paiement, un horodatage et le hash du bloc précédent. Modifier ce bloc invaliderait toute la chaîne.
Block #N | data: {membre, montant, payMode} | prevHash: hash(N-1) | hash: sha256(data+prevHash)
3
Le vote est cryptographiquement sécurisé
Chaque vote est signé par la clé privée du membre et enregistré dans un bloc dédié. Le contrat vérifie que le membre n'a pas déjà voté (mapping aVote). Impossible de voter deux fois ou de modifier un vote.
require(!aVote[msg.sender], "Déjà voté") → vote immuable
4
Smart contracts : automatisation sans intermédiaire
Les règles (quorum, objectif, libération des fonds) sont codées dans des smart contracts Solidity. Elles s'exécutent automatiquement dès que les conditions sont remplies. Aucun administrateur ne peut bloquer ni détourner.
if (totalVotes >= quorum) → autoriserAchat() automatique
5
Transparence totale pour tous les membres
Chaque membre peut vérifier l'intégrité complète de la chaîne depuis son téléphone. Le hash de chaque bloc est recalculable publiquement. Toute tentative de fraude est immédiatement détectable.
hash(block_N) = sha256(data_N + prevHash_{N-1}) — public et vérifiable
